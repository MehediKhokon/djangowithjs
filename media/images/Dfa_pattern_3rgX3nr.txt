#include <iostream>
#define EOS '\0'
using namespace std;

int main()
{
    cout<< "Enter the String abc*d"<<endl;
    string str;
    cin>>str;
    int state;
    state = 1;
    int i = 0;
    while(str[i] != EOS){
        if(state == 1 && str[i] == 'a'){
            state = 2;
            i++;
        }
        else if(state == 2 && str[i] == 'b'){
            state = 3;
            i++;
        }
        else if(state == 3 && str[i] == 'c'){
            state = 3;
            i++;
        }
        else if(state == 3 && str[i] == 'd'){
            state = 4;
            i++;
        }
        else{
            state = 5;
            break;
        }
        
      
    }
    if( state == 4){
        cout<<"valid"<<endl;
    }
    else{
        cout<<"invalid";
    }
    
    

    return 0;
}
